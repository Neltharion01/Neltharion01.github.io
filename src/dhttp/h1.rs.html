13fn parse_ver(ver: &str) -> Option<HttpVersion> {
14    let mut split = ver.strip_prefix("HTTP/")?.split('.');
15    let major = split.next()?.parse().ok()?;
16    let minor = split.next()?.parse().ok()?;
17    if split.next().is_some() {
18        // 3rd element is invalid
19        return None;
20    }
21    Some(HttpVersion { major, minor })
22}
23
24fn parse_header(header: &str) -> Option<HttpHeader> {
25    let colon = header.find(':')?;
26    let name = header[..colon].to_string();
27    let value = header[colon+1..].trim().to_string();
28    Some(HttpHeader { name, value })
29}

31fn split3(line: &str) -> Option<(&str, &str, &str)> {
32    let mut split = line.split_whitespace();
33    let (method, route, version) = (split.next()?, split.next()?, split.next()?);
34    if split.next().is_some() {
35        // 4th element is invalid
36        return None;
37    }
38    Some((method, route, version))
39}

41/// Reads a request from the provided stream
42pub(crate) async fn read(conn: impl HttpRead) -> Result<HttpRequest, HttpRequestError> {
43    let mut lines = conn.lines();
44
45    // get first line
46    let first = lines.next_line().await?.ok_or(HttpRequestError::EarlyEof)?;
47    // and slice it by 3 components
48    let (method, route, version) = split3(&first).ok_or(HttpRequestError::InvalidPrelude)?;
49    // then parse method, allocate route, parse version
50    let method = HttpMethod::new(method);
51    let route = route.to_string();
52    let version = parse_ver(version).ok_or(HttpRequestError::InvalidVersion)?;
53    // read headers
54    let mut headers = vec![];
55    loop {
56        // will return if connection is shut down without \n\n
57        let line = lines.next_line().await?.ok_or(HttpRequestError::EarlyEof)?;
58        if line.is_empty() {
59            // empty line = end of request
60            break;
61        }
62        headers.push(parse_header(&line).ok_or(HttpRequestError::InvalidHeader)?);
63    }
64
65    let addr = IpAddr::V4(Ipv4Addr::UNSPECIFIED);
66    let mut req = HttpRequest { method, route, version, headers, len: 0, addr };
67
68    if let Some(content_length) = req.get_header("Content-Length") {
69        req.len = content_length.parse().map_err(|_| HttpRequestError::InvalidLength)?;
70    }
71
72    Ok(req)
73}

75/// Send the request
76pub(crate) async fn send(req: &HttpRequest, res: HttpResponse, conn: &mut dyn HttpConnection) -> io::Result<()> {
77    let code = res.code;
78    let status = code.as_str();
79    let mut buf = format!("HTTP/1.1 {code} {status}\r\n").into_bytes();
80
81    for header in &res.headers {
82        write!(&mut buf, "{}: {}\r\n", &header.name, &header.value).unwrap();
83    }
84
85    if !res.content_type.is_empty() {
86        write!(&mut buf, "Content-Type: {}\r\n", &res.content_type).unwrap();
87    }
88
89    match &res.body {
90        HttpBody::Bytes(bytes) => write!(&mut buf, "Content-Length: {}\r\n", bytes.len()).unwrap(),
91        HttpBody::File { len, .. } => write!(&mut buf, "Content-Length: {}\r\n", len).unwrap(),
92        HttpBody::Empty | HttpBody::Upgrade(_) => {},
93    };
94    buf.extend(b"\r\n");
95
96    if let HttpBody::Bytes(bytes) = &res.body {
97        buf.extend(bytes);
98    }
99
100    // Send static part
101    conn.write_all(&buf).await?;
102
103    // Don't send body on head requests
104    if req.method == HttpMethod::Head { return Ok(()); }
105
106    // Now, handle the body
107    match res.body {
108        HttpBody::Empty | HttpBody::Bytes(_) => {},
109        HttpBody::File { file, len } => {
110            tokio::io::copy(&mut file.take(len), conn).await?;
111        }
112        HttpBody::Upgrade(mut handler) => {
113            handler.upgrade_raw(conn).await?;
114            conn.shutdown().await?;
115        }
116    }
117
118    Ok(())
119}

121/// Error when parsing an HTTP/1.1 request.
122/// For debugging purposes only
123#[derive(Debug)]
124pub(crate) enum HttpRequestError {
125    /// IO error
126    Io(io::Error),
127    /// Some part of request contained invalid Unicode
128    NotUnicode(FromUtf8Error),
129    /// Request exceed its size limit
130    TooLong,
131    /// Request ended too early
132    EarlyEof,
133    /// First line didn't follow the `METHOD /route HTTP/1.1` format
134    InvalidPrelude,
135    /// Could not parse HTTP version
136    InvalidVersion,
137    /// Header line did not contain a colon
138    InvalidHeader,
139    /// `Content-Length` header did not contain a number
140    InvalidLength,
141}
142
143impl fmt::Display for HttpRequestError {
144    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
145        match self {
146            HttpRequestError::Io(err) => err.fmt(fmt),
147            HttpRequestError::NotUnicode(err) => err.fmt(fmt),
148            HttpRequestError::TooLong => fmt.write_str("request too long"),
149            HttpRequestError::EarlyEof => fmt.write_str("incomplete request"),
150            // first line of request did not contain exactly 3 elements (method, path and version)
151            HttpRequestError::InvalidPrelude => fmt.write_str("invalid prelude"),
152            HttpRequestError::InvalidVersion => fmt.write_str("invalid http version"),
153            HttpRequestError::InvalidHeader => fmt.write_str("header without a colon"),
154            HttpRequestError::InvalidLength => fmt.write_str("content-length header did not contain a number"),
155        }
156    }
157}
158
159impl From<FromUtf8Error> for HttpRequestError {
160    fn from(err: FromUtf8Error) -> HttpRequestError {
161        HttpRequestError::NotUnicode(err)
162    }
163}
164
165impl From<io::Error> for HttpRequestError {
166    fn from(err: io::Error) -> HttpRequestError {
167        if err.kind() == ErrorKind::UnexpectedEof {
168            HttpRequestError::TooLong
169        } else {
170            HttpRequestError::Io(err)
171        }
172    }
